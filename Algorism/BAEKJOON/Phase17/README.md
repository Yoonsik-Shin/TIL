a = [1, 3, 2, 6, 2]
p = [1, 4, 6, 12, 14] 
'''
p[0] = a[0]
p[1] = a[1] + p[0]
p[2] = a[2] + p[2]
p[n] = a[n] + p[n-1] ✔️
'''
0 ~ n번 index
ans : p[n]

i ~ j번 index
ans : p[j] - p[i-1]
